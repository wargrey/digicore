% order is important for `tcbox`
% so \usepackage{tcolorbox} has been moved into `literacy.tex`

\newcommand{\SharpBox}[4]{
   \begin{tcolorbox}
   [enhanced, breakable, nobeforeafter,
      sharp corners=#1, colframe=#2, colback=#3,
      boxsep=1mm, left=0mm, right=0mm, top=0mm, bottom=0mm]
   #4
   \end{tcolorbox}}

\newcommand{\RoundedBox}[3]{
   \begin{tcolorbox}
   [enhanced, breakable, nobeforeafter,
      colframe=#1, colback=#2,
      boxsep=1mm, left=0mm, right=0mm, top=0mm, bottom=0mm]
   #3
   \end{tcolorbox}}

\newcommand{\NorthBox}[2]{
   \begin{tcolorbox}
   [enhanced, breakable,
      sharp corners=north,
      colback=#1, colframe=#1,
      borderline north={.33pt}{-.5pt}{MidnightBlue!50}]
   #2
   \end{tcolorbox}}

\newcommand{\NorthBoxWithTitleAtRight}[4]{
   \begin{tcolorbox}
   [enhanced, breakable,
      sharp corners=north,
      colback=#1, colframe=#1,
      colbacktitle=#2,
      borderline north={.33pt}{-.5pt}{MidnightBlue!50},
      attach boxed title to top right={ xshift=-1.618em, yshift=-2mm },
      title={#3}]
   #4
   \end{tcolorbox}}

\renewcommand{\boxed}[1]{\NorthBox{AliceBlue}{#1}}
\newcommand{\boxedFile}[2]{\NorthBoxWithTitleAtRight{AliceBlue}{DodgerBlue}{#1}{#2}}
